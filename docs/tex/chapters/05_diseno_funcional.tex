\section{Dominio Archivo Institucional}
\subsection{Capacidades funcionales}
\begin{itemize}
  \item Busqueda por texto y metadatos.
  \item Filtros por tipologia, fecha y tesauro.
  \item Vista de resultados con navegacion paginada.
  \item Consulta de detalle documental en modal.
  \item Exportacion de reportes filtrados.
\end{itemize}

\subsection{Flujo funcional de consulta}
\begin{enumerate}
  \item El usuario define criterios de busqueda.
  \item El sistema ejecuta filtrado y ordenamiento.
  \item Se presentan resultados trazables con metadatos clave.
  \item El usuario puede abrir detalle o exportar vista.
\end{enumerate}

\section{Dominio RRHH}
\subsection{Capacidades funcionales}
\begin{itemize}
  \item Expediente unificado por persona.
  \item Filtros por tipologia, estado, fecha y personas relacionadas.
  \item Dossier integral con contexto administrativo.
  \item Exportacion de expediente consolidado.
\end{itemize}

\subsection{Flujo funcional de expediente}
\begin{enumerate}
  \item Seleccion de persona o criterio documental.
  \item Construccion de expediente y categorias asociadas.
  \item Visualizacion integral de archivos vinculados.
  \item Generacion de reporte de expediente.
\end{enumerate}

\section{Panel Administrativo}
\begin{itemize}
  \item Control de registros, categorias y usuarios.
  \item Vista de indicadores y estadisticas de operacion.
  \item Herramientas de gobierno documental.
\end{itemize}
